Measure distance

Build a basic project with ultra soni sensor. This the first time using a ultra sonic sensor.

Components

  • Breadboard
  • Jumperwires
  • Arduino uno/Arduino mega/Arduino due
  • Usb cable
  • ultra sonic sensor

Schematics

Description

The ultra sonic sensor look like a face but it is used for measuring distance of an object. The sensor can olny measure up to 400cm and start at 2cm.

Code 

          
            int trigPin = 10;
            int echoPin = 13;
            
            void setup(){
              Serial.begin(9600);
              pinMode(trigPin, OUTPUT);
              pinMode(echoPin,INPUT);
              
            }
            
            void loop(){
              digitalWrite(trigPin, LOW);
              delayMicroseconds(2);
            
              digitalWrite(trigPin, HIGH);
              delayMicroseconds(10);
              digitalWrite(trigPin, LOW);
            
              float duration = pulseIn(echoPin, HIGH);
              float distance = (duration/2)* 0.0344;
            
              if(distance >= 400 ||distance <= 2) {
                Serial.print("Distance =");
                Serial.println("out of range");
              }
            
              else{
                Serial.print("Distance =");
                Serial.print(distance);
                Serial.println(" cm");
                delay(500);
              }
            
              delay(500);
            }
